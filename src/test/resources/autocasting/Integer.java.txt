package autocasting;

class CharAutocasting {

	public int integer(){
	    Integer val = 110;
        Integer res = minus(val, val / 2);
        if(res != null){
            return res + 5;
        }

        return 0;
    }

    private int minus(int arg1, int arg2){
        Integer result = arg1 - arg2;
        int test = result + 4;
        if(test < 0)
            return 0;

        return result;
    }
}
